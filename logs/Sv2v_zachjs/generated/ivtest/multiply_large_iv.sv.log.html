	a,
	b,
	sum
);
	parameter N = 1;
	input [N - 1:0] a;
	input [N - 1:0] b;

	integer i;
	always @(a or b) begin
		tmp = 0;
		for (i = 0; i < N; i = i + 1)
			if (b[i])

	end
