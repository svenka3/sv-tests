module stim (
	clk,
	reset,
	count,
	en
);
	input clk;
	input reset;
	output reg [2:0] count;
	output reg en;
	always @(posedge clk)
		if (reset)
			count <= 3'b000;
		else
			count <= count + 1;

		en = 1;

			;
		en = 0;
	end
