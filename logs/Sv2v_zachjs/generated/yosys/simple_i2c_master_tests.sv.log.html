);

	input cmd;
	output reg cmd_stop;

	always @(posedge clk)

			if (~slave_wait) begin

			end

			end
		end

			cnt <= #(1) cnt - 16'h0001;
			clk_en <= #(1) 1'b0;
		end

		if (clk_en)
			cmd_stop <= #(1) cmd;
