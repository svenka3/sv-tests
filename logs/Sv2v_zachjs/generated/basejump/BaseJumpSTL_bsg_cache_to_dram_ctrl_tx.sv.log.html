	dma_data_v_i,
	dma_data_yumi_o,

	app_wdf_data_o,
	app_wdf_mask_o,
	app_wdf_end_o,
	app_wdf_rdy_i
);
	parameter _sv2v_width_num_cache_p = 24;
	parameter [_sv2v_width_num_cache_p - 1:0] num_cache_p = "inv";
	parameter _sv2v_width_data_width_p = 24;
	parameter [_sv2v_width_data_width_p - 1:0] data_width_p = "inv";
	parameter _sv2v_width_block_size_in_words_p = 24;
	parameter [_sv2v_width_block_size_in_words_p - 1:0] block_size_in_words_p = "inv";
	parameter _sv2v_width_dram_ctrl_burst_len_p = 24;
	parameter [_sv2v_width_dram_ctrl_burst_len_p - 1:0] dram_ctrl_burst_len_p = "inv";
	localparam mask_width_lp = data_width_p >> 3;
	localparam num_req_lp = block_size_in_words_p / dram_ctrl_burst_len_p;
	localparam lg_num_cache_lp = (num_cache_p == 1 ? 1 : $clog2(num_cache_p));
	localparam lg_dram_ctrl_burst_len_lp = (dram_ctrl_burst_len_p == 1 ? 1 : $clog2(dram_ctrl_burst_len_p));
	input clk_i;
	input reset_i;
	input v_i;
	input [lg_num_cache_lp - 1:0] tag_i;
	output wire ready_o;
	input [(num_cache_p * data_width_p) - 1:0] dma_data_i;
	input [num_cache_p - 1:0] dma_data_v_i;
	output wire [num_cache_p - 1:0] dma_data_yumi_o;
	output wire app_wdf_wren_o;
	output wire [data_width_p - 1:0] app_wdf_data_o;
	output wire [mask_width_lp - 1:0] app_wdf_mask_o;
	output reg app_wdf_end_o;
	input app_wdf_rdy_i;
	wire [lg_num_cache_lp - 1:0] tag_fifo_data_lo;
	wire tag_fifo_v_lo;
	reg tag_fifo_yumi_li;
	bsg_fifo_1r1w_small #(
		.width_p(lg_num_cache_lp),
		.els_p(num_cache_p * num_req_lp)

		.reset_i(reset_i),
		.v_i(v_i),
		.data_i(tag_i),
		.ready_o(ready_o),
		.v_o(tag_fifo_v_lo),
		.data_o(tag_fifo_data_lo),
		.yumi_i(tag_fifo_yumi_li)
	);
	wire [num_cache_p - 1:0] cache_sel;
	bsg_decode_with_v #(.num_out_p(num_cache_p)) demux(
		.i(tag_fifo_data_lo),
		.v_i(tag_fifo_v_lo),
		.o(cache_sel)
	);
	assign dma_data_yumi_o = (cache_sel & dma_data_v_i) & {num_cache_p {app_wdf_rdy_i}};

		.clear_i(clear_li),

		.count_o(count_lo)
	);
