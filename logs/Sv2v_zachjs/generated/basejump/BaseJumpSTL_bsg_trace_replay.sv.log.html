module bsg_trace_replay (
	clk_i,
	reset_i,
	en_i,
	v_i,
	data_i,
	ready_o,
	v_o,
	data_o,
	yumi_i,
	rom_addr_o,
	rom_data_i,
	done_o,
	error_o
);
	parameter payload_width_p = 80;
	parameter rom_addr_width_p = 6;
	parameter counter_width_p = (payload_width_p < 16 ? payload_width_p : 16);
	parameter debug_p = 1;
	parameter finish_on_error_p = 1;
	parameter opcode_width_lp = 4;
	input clk_i;
	input reset_i;
	input en_i;
	input v_i;
	input [payload_width_p - 1:0] data_i;
	output reg ready_o;
	output reg v_o;
	output wire [payload_width_p - 1:0] data_o;
	input yumi_i;
	output [rom_addr_width_p - 1:0] rom_addr_o;
	input [payload_width_p + 3:0] rom_data_i;
	output wire done_o;
	output wire error_o;
	reg [counter_width_p - 1:0] cycle_ctr_r;
	reg [counter_width_p - 1:0] cycle_ctr_n;
	reg [rom_addr_width_p - 1:0] addr_r;
	wire [rom_addr_width_p - 1:0] addr_n;
	reg done_r;
	reg done_n;
	reg error_r;
	reg error_n;
	assign rom_addr_o = addr_r;
	assign data_o = rom_data_i[0+:payload_width_p];
	assign done_o = done_r;
	assign error_o = error_r;
	always @(posedge clk_i)
		if (reset_i) begin
			addr_r <= 0;
			done_r <= 0;
			error_r <= 0;
			cycle_ctr_r <= 16'b0000000000000001;
		end
		else begin
			addr_r <= addr_n;
			done_r <= done_n;
			error_r <= error_n;
			cycle_ctr_r <= cycle_ctr_n;
		end
	wire [3:0] op;
	assign op = rom_data_i[payload_width_p+:4];
	reg instr_completed;
	assign addr_n = (instr_completed ? addr_r + 1'b1 : addr_r);
	localparam [opcode_width_lp - 1:0] eDone = 4'd3;
	localparam [opcode_width_lp - 1:0] eReceive = 4'd2;
	localparam [opcode_width_lp - 1:0] eSend = 4'd1;
	always @(*) begin
		v_o = 1'b0;
		ready_o = done_r;
		done_n = done_r;
		if ((!done_r & en_i) & ~reset_i)
			case (op)
				eSend: v_o = 1'b1;
				eReceive: ready_o = 1'b1;
				eDone: done_n = 1'b1;
				default:
					;
			endcase
	end
	localparam [opcode_width_lp - 1:0] eCycleDec = 4'd5;
	localparam [opcode_width_lp - 1:0] eCycleInit = 4'd6;
	localparam [opcode_width_lp - 1:0] eFinish = 4'd4;
	localparam [opcode_width_lp - 1:0] eNop = 4'd0;
	always @(*) begin
		instr_completed = 1'b0;
		error_n = error_r;
		cycle_ctr_n = cycle_ctr_r;
		if ((!done_r & en_i) & ~reset_i)
			case (op)
				eNop: instr_completed = 1'b1;
				eSend:
					if (yumi_i)
						instr_completed = 1'b1;
				eReceive:
					if (v_i) begin
						instr_completed = 1'b1;
						if (error_r == 0)
							error_n = data_i != data_o;
					end
				eDone: instr_completed = 1'b1;
				eFinish: instr_completed = 1'b1;
				eCycleDec: begin
					cycle_ctr_n = cycle_ctr_r - 1'b1;
					instr_completed = ~(|cycle_ctr_r);
				end
				eCycleInit: begin
					cycle_ctr_n = rom_data_i[counter_width_p - 1:0];
					instr_completed = 1'b1;
				end
				default: begin
					error_n = 1'b1;
					instr_completed = 1'b1;
				end
			endcase
	end
	always @(negedge clk_i)
		if ((instr_completed & ~reset_i) & ~done_r)
			case (op)
				eNop:
					;
				eSend:
					if (debug_p >= 2)
						$display("### bsg_trace_replay SEND %d'b%b (%m)", payload_width_p, data_o);
				eReceive:
					if (data_i !== data_o) begin
						$display("############################################################################");
						$display("### bsg_trace_replay RECEIVE unmatched (%m) ");
						$display("###    ");
						$display("### FAIL (trace mismatch) = %h", data_i);
						$display("###              expected = %h\n", data_o);
						$display("############################################################################");
						if (finish_on_error_p == 1)
							$finish;
					end
					else if (debug_p >= 2)
						$display("### bsg_trace_replay RECEIVE matched %h (%m)", data_o);
				eDone:
					if (debug_p >= 1) begin
						$display("############################################################################");
						$display("###### bsg_trace_replay DONE done_o=1 (trace finished addr=%x) (%m)", rom_addr_o);
						$display("############################################################################");
					end
				eFinish: begin
					if (debug_p >= 1) begin
						$display("############################################################################");
						$display("###### bsg_trace_replay FINISH (trace finished; CALLING $finish) (%m)");
						$display("############################################################################");
					end
					$finish;
				end
				eCycleDec:
					if (debug_p >= 2)
						$display("### bsg_trace_replay CYCLE DEC cycle_ctr_r = %x (%m)", cycle_ctr_r);
				eCycleInit:
					if (debug_p >= 2)
						$display("### bsg_trace_replay CYCLE INIT = %x (%m)", cycle_ctr_n);
				default: begin
					$error("### bsg_trace_replay UNKNOWN op %x (%m)\n", op);
					if (finish_on_error_p == 1)
						$finish;
				end
			endcase
