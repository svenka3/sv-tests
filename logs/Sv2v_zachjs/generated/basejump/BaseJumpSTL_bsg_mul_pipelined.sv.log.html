module bsg_mul_SDN (
	x_i,
	signed_i,
	SDN_o
);
	parameter width_p = 16;
	parameter rows_lp = (width_p / 2) + 1;
	input [width_p - 1:0] x_i;
	input signed_i;
	output [(rows_lp * 3) - 1:0] SDN_o;
	wire [width_p + 2:0] temp_x = {2'b00, x_i, 1'b0};
	genvar i;
	generate
		for (i = 0; i < rows_lp; i = i + 1) begin : rof
			wire [2:0] trip;
			if (i != (rows_lp - 1)) begin : fi
				assign trip = temp_x[(2 * i) + 2:2 * i];
			end
			else begin : fi
				assign trip = temp_x[(2 * i) + 2:2 * i] & ~{3 {signed_i}};
			end
			assign SDN_o[i * 3] = trip[2];
			assign SDN_o[(i * 3) + 1] = (&trip[1:0] & ~trip[2]) | (~(|trip[1:0]) & trip[2]);
			assign SDN_o[(i * 3) + 2] = trip[0] ^ trip[1];
		end
	endgenerate
endmodule

module bsg_mul_green_booth_dots (
	SDN_i,
	y_i,
	dot_o
);
	parameter _sv2v_width_width_p = 24;
	parameter [_sv2v_width_width_p - 1:0] width_p = "inv";
	parameter harden_p = 0;
	parameter _sv2v_width_blocks_p = 24;
	parameter [_sv2v_width_blocks_p - 1:0] blocks_p = "inv";
	parameter _sv2v_width_group_vec_p = 24;
	parameter [_sv2v_width_group_vec_p - 1:0] group_vec_p = "inv";
	input [5:0] SDN_i;
	input [width_p - 1:0] y_i;
	output [width_p + 1:0] dot_o;
	wire [width_p + 1:0] y_pad = {y_i, 2'b00};
	genvar i;
	genvar j;
	generate
		for (j = 0; j < blocks_p; j = j + 1) begin : blk
			localparam group_end_lp = (group_vec_p >> ((j + 1) << 3)) & 8'hff;
			localparam group_start_lp = (group_vec_p >> (j << 3)) & 8'hff;
			localparam [31:0] blocks_lp = group_end_lp - group_start_lp;
			if ((j != 0) && harden_p) begin : macro
				bsg_and #(
					.width_p(blocks_lp),
					.harden_p(harden_p)
				) ba(
					.a_i({blocks_lp {SDN_i[5]}}),
					.b_i(y_pad[group_start_lp+:blocks_lp]),
					.o(dot_o[group_start_lp+:blocks_lp])
				);
			end
			else begin : notmacro
				for (i = 0; i < blocks_lp; i = i + 1) begin : b
					if ((j == 0) && (i == 0)) begin : fi
						assign dot_o[0] = SDN_i[0];
					end
					else if ((j == 0) && (i == 1)) begin : fi
						assign dot_o[1] = 1'b0;
					end
					else begin : fi
						assign dot_o[group_start_lp + i] = SDN_i[5] & y_pad[group_start_lp + i];
					end
				end
			end
		end
	endgenerate
endmodule
