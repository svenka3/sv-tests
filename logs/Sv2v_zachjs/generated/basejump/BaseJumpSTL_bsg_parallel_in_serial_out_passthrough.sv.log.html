	reset_i,
	v_i,
	data_i,
	ready_and_o,
	v_o,
	data_o,
	ready_and_i
);

	parameter els_p = -1;
	parameter hi_to_lo_p = 0;
	input clk_i;
	input reset_i;
	input v_i;
	input [(els_p * width_p) - 1:0] data_i;
	output ready_and_o;
	output v_o;
	output [width_p - 1:0] data_o;
	input ready_and_i;
	wire [els_p - 1:0] count_r;
	generate

			assign ready_and_o = ready_and_i;
		end

			wire last_word;
			bsg_counter_clear_up_one_hot #(
				.max_val_p(els_p - 1),
				.init_val_p(1)
			) counter(
				.clk_i(clk_i),
				.reset_i(reset_i),
				.clear_i(ready_and_o & v_i),
				.up_i((v_o & ready_and_i) & ~last_word),
				.count_r_o(count_r)
			);

			assign ready_and_o = ready_and_i & last_word;
		end
	endgenerate

	generate

			assign data_li = data_i;
		end
		else begin : hi2lo

				.width_p(width_p),
				.els_p(els_p)
			) bar(
				.i(data_i),
				.o(data_li)
			);
		end
	endgenerate

		.width_p(width_p),
		.els_p(els_p)
